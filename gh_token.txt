#register autograder
curl --location 'http://dl-berlin.ecn.purdue.edu/api/register' \
--header 'Content-Type: application/json' \
--data '{
    "group": 13,
    "github": "link_to_your_github_repo",
    "names": [
            "Dylan Manning", 
            "Cecilia Jiang", 
            "Tianyu Zhang", 
            "Ava Lyall ",
    ],
    "gh_token": "your_github_token"
}'


#schedule run
curl --location 'http://dl-berlin.ecn.purdue.edu/api/schedule' \
--header 'Content-Type: application/json' \
--data '{
    "group": 13,
    "gh_token": "your_github_token"
}'


#You can monitor the progress by looking at /run/all endpoint, this shows all the scheduled task queues.
curl --location --request GET 'http://dl-berlin.ecn.purdue.edu/api/run/all' \
--header 'Content-Type: application/json' \
--data '{
    "group": your_group_number,
    "gh_token": "your_github_token"
}'

#You can get the best score at /best_run and the latest score at /last_run.
#Best Run:
curl --location --request GET 'http://dl-berlin.ecn.purdue.edu/api/best_run' \
--header 'Content-Type: application/json' \
--data '{
    "group": your_group_number,
    "gh_token": "your_github_token"
}'
#Last Run:

curl --location --request GET 'http://dl-berlin.ecn.purdue.edu/api/last_run' \
--header 'Content-Type: application/json' \
--data '{
    "group": your_group_number,
    "gh_token": "your_github_token"
}'


#You can also download the log file by passing the log URL to the /log/download endpoint. (The sysout download is only available for the CLI checker phase)
curl --location --request GET 'http://dl-berlin.ecn.purdue.edu/api/log/download' \
--header 'Content-Type: application/json' \
--data '{
    "group": your_group_number,
    "gh_token": "your_github_token",
    "log": "path/to/.log (copy-paste from last/best run output)"
}'
